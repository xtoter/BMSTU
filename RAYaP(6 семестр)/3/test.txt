(procedure print_number (x)
    (if ((L x) "<" 0)
        (call out '-')
        (pcall print_number ("-" (L x)))
        (preturn)
    )
    (if ((L x) "<" 10)
        (call out ('0' "+" (L x)))
        (preturn)
    )
    (pcall print_number ((L x) "/" 10))
    (call out ('0' "+" ((L x) "%" 10)))
    (preturn)
)


(procedure print_number_n (x)
    (pcall print_number (L x))
    (call newline)
    (preturn)
)

(function gcd (x y)
    (var (rem 1))
    (while ((L y) "<>" 0)
        (rem "=" ((L x) "%" (L y)))
        (x "=" (L y))
        (y "=" (L rem))
    )
    (return (L x))
)
