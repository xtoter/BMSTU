grammar  ->  rule { rule }
rule  ->  Nonterminal  "(" expression ")"
expression  ->  term { "," term }
term  ->  factor { factor }
factor  ->  Terminal | NonTerminal| option |grouping
option  ->  "{" expression "}"
grouping -> "(" expression ")"


/* варианты разделяются запятыми */
E  ( T {("+","-") T} )
T  ( F {("*","/") F} )
F  ( "n", "-" F,
     "(" E ")" )
