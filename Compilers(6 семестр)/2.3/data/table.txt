Grammar -> Rule Grammar|
Rule -> Nonterm Exprbracket | Axiom Nonterm Exprbracket
Exprbracket -> ( Expr) Exprbracket |
Expr -> Operation Nonterms | Nonterms | NestedNonTerm | Terminal
NestedNonTerm -> "(" NonTerm ")"
Nonterms -> Nonterm Nonterm
Operation -> "+" | "*"
Terminal -> " a-z "
Axiom -> *
Nonterm -> A-Z
